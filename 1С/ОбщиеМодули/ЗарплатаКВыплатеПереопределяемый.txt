Процедура ПровестиПоХарактеруВыплаты(Документ, Отказ) Экспорт
	
	Для Каждого СтрокаЗарплаты Из Документ.Зарплата Цикл
		
		Движение = Документ.Движения.ЗарплатаЗаМесяц.ДобавитьРасход();
		
		// свойства
		Движение.Период			= КонецМесяца(Документ.Дата);
		
		// измерения
		Движение.Физлицо		= СтрокаЗарплаты.Физлицо;
		
		// ресурсы
		Движение.СуммаУпр		= СтрокаЗарплаты.Сумма;

		// реквизиты
		Движение.ХарактерВыплаты= Документ.ХарактерВыплаты;
		
	КонецЦикла
	
КонецПроцедуры	

// Процедура осуществляет регистрацию взаиморасчетов документа при упрощенном учете взаиморасчетов
//
// Параметр:
//   Документ - документ "Зарплата к выплате организаций"
//   Отказ    - признак отказа от проведения
//
Процедура ПровестиВзаиморасчеты(Документ, Отказ) Экспорт
	
	Для Каждого СтрокаЗарплаты Из Документ.Зарплата Цикл
		
		СтрокаДвижения = Документ.Движения.ВзаиморасчетыСРаботниками.ДобавитьРасход();
		
		// свойства
		СтрокаДвижения.Период			= КонецМесяца(Документ.Дата);
		
		// измерения
		СтрокаДвижения.Физлицо		= СтрокаЗарплаты.Физлицо;
		
		// ресурсы
		СтрокаДвижения.СуммаУпр		= СтрокаЗарплаты.Сумма;

		// реквизиты
		СтрокаДвижения.ХарактерВыплаты= Документ.ХарактерВыплаты;
		
	КонецЦикла
	
КонецПроцедуры
