// Возвращает список видов документов для этого журнала в виде списка значений
// 
Функция ПолучитьСписокВидовДокументов(мСписокВидовДокументов) Экспорт
	Если мСписокВидовДокументов <> НеОпределено Тогда
		Возврат мСписокВидовДокументов;
	КонецЕсли;
	
	мСписокВидовДокументов = Новый СписокЗначений;
	мСписокВидовДокументов.Добавить("НачислениеПоБольничномуЛисту");
	мСписокВидовДокументов.Добавить("НачислениеОтпускаРаботникамОрганизаций");
	мСписокВидовДокументов.Добавить("ОплатаПоСреднемуЗаработку");
	мСписокВидовДокументов.Добавить("НевыходыВОрганизациях");
	
	РаботаСДиалогамиЗК.ДополнитьСписокВидовДокументов(Метаданные.ЖурналыДокументов.РасчетОплатыНеявок, мСписокВидовДокументов);
	
	Возврат мСписокВидовДокументов;
	
КонецФункции
