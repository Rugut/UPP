// Возвращает список видов документов для этого журнала в виде списка значений
// 
Функция ПолучитьСписокВидовДокументов(мСписокВидовДокументов) Экспорт
	Если мСписокВидовДокументов <> НеОпределено Тогда
		Возврат мСписокВидовДокументов;
	КонецЕсли;
	
	мСписокВидовДокументов = Новый СписокЗначений;
	мСписокВидовДокументов.Добавить("НачислениеЗарплатыРаботникамОрганизаций");
	мСписокВидовДокументов.Добавить();
	мСписокВидовДокументов.Добавить("ПремииРаботниковОрганизаций");
	мСписокВидовДокументов.Добавить("РегистрацияРазовыхНачисленийРаботниковОрганизаций");
	мСписокВидовДокументов.Добавить("РегистрацияРазовыхУдержанийРаботниковОрганизаций");
	мСписокВидовДокументов.Добавить();
	
	РаботаСДиалогамиЗК.ДополнитьСписокВидовДокументов(Метаданные.ЖурналыДокументов.ДокументыПоНачислениюЗарплатыОрганизаций, мСписокВидовДокументов);
	
	Возврат мСписокВидовДокументов;
	
КонецФункции
