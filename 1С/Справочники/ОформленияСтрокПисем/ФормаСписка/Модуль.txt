Процедура СправочникСписокПриВыводеСтроки(Элемент, ОформлениеСтроки, ДанныеСтроки)
	
	ЦветТекста = ДанныеСтроки.ЦветТекста.Получить();
	Если ТипЗнч(ЦветТекста) = Тип("Цвет") Тогда
		ОформлениеСтроки.ЦветТекста = ЦветТекста;
	КонецЕсли;
	
	ЦветФона = ДанныеСтроки.ЦветФона.Получить();
	Если ТипЗнч(ЦветФона) = Тип("Цвет") Тогда
		ОформлениеСтроки.ЦветФона = ЦветФона;
	КонецЕсли;
	
	Шрифт = Новый Шрифт(,,ДанныеСтроки.Жирный,ДанныеСтроки.Наклонный,ДанныеСтроки.Подчеркнутый,ДанныеСтроки.Зачеркнутый);
	ОформлениеСтроки.Шрифт = Шрифт;
	
КонецПроцедуры

Процедура ПриЗакрытии()
	
	Если ТипЗнч(ВладелецФормы) = Тип("Форма") Тогда // Оповещение формы менеджера контактов
		ОповеститьОЗаписиНовогоОбъекта(Справочники.ОформленияСтрокПисем.ПустаяСсылка());
	КонецЕсли; 
	
КонецПроцедуры
