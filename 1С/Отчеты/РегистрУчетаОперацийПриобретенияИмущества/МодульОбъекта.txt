	НайденныйВидРасхода = ВидыРасходов.НайтиПоЗначению(ВидРасхода);

	Если НайденныйВидРасхода <> Неопределено Тогда

			Соответствие = ИСТИНА;

	КонецЕсли;

	Возврат Соответствие;
