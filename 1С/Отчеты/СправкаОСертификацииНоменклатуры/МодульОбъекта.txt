							СтСерт = ?(СтСерт = "",СтСерт,СтСерт + Символы.ПС) + ДатаСерт;

							СтОтказ = ?(СтОтказ = "",СтОтказ,СтОтказ + Символы.ПС) + ДатаОтказ;

							СтрокаОтказ = ТабОтказ.Добавить();

					ПерСерт = 1 + (НачалоДня(МинДатаСерт) - НачалоДня(МинДатаЗаявки)) / 86400;
