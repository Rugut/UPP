// Процедура сдвигает текущую строку дерева в коллекции на Сдвиг
//
// Параметры
//	Сдвиг - количество позиций сдвига
//
Процедура СдвинутьТекущуюСтроку(Сдвиг)

	Если ЭлементыФормы.Список.ТекущаяСтрока <> Неопределено Тогда

		Если Список.Индекс(ЭлементыФормы.Список.ТекущаяСтрока) + Сдвиг >= 0
			И Список.Индекс(ЭлементыФормы.Список.ТекущаяСтрока) + Сдвиг <= Список.Количество()-1 Тогда
			
			Список.Сдвинуть(ЭлементыФормы.Список.ТекущаяСтрока, Сдвиг);
		КонецЕсли;
	КонецЕсли;

КонецПроцедуры // СдвинутьТекущуюСтроку()
