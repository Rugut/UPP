Процедура СписокКодовВыбор(Элемент, ВыбраннаяСтрока, Колонка, СтандартнаяОбработка)
	
	Закрыть(ВыбраннаяСтрока);
	
КонецПроцедуры

Процедура ДействияФормыВыбрать(Кнопка)
	
	ВыбраннаяСтрока = ЭлементыФормы.СписокКодов.ТекущаяСтрока;
	
	Если НЕ (ВыбраннаяСтрока = Неопределено) Тогда
		Закрыть(ВыбраннаяСтрока);
	КонецЕсли;
	
КонецПроцедуры

Процедура ПриОткрытии()
	Если КодИсходнойСтроки <> "" Тогда
		ЭлементыФормы.СписокКодов.ТекущаяСтрока = СписокКодов.Найти(КодИсходнойСтроки, "Код");
	КонецЕсли;
КонецПроцедуры

Процедура ПередОткрытием(Отказ, СтандартнаяОбработка)
	
	МинимальнаяШирина = 4;
	
	ДлинаКода = 0;
	Для Каждого ЭлементСписка Из СписокКодов Цикл
		ДлинаКодаЭлемента = СтрДлина(ЭлементСписка.Код);
		ДлинаКода = Макс(ДлинаКода, ДлинаКодаЭлемента);
	КонецЦикла;
	
	НормированнаяШирина = ДлинаКода + Окр(ДлинаКода/10, 0);
	НормированнаяШирина = Макс(НормированнаяШирина, МинимальнаяШирина);
	
	ЭлементыФормы.СписокКодов.Колонки.Код.Ширина = НормированнаяШирина;
	
КонецПроцедуры
