Процедура ОбработкаОповещения(ИмяСобытия, Параметр, Источник)
	
	Если ИмяСобытия = "Чтение запроса" И ТипЗнч(Параметр) = Тип("ДокументСсылка.ЗапросНаИнформационноеОбслуживаниеНалогоплательщика") Тогда
		ЭлементыФормы.ДокументСписок.ОбновитьСтроки();
	ИначеЕсли ИмяСобытия = "Опрос форм" И Параметр.Ключ = "Документ_ЗапросНаИнформационноеОбслуживаниеНалогоплательщика_ФормаСписка" Тогда
		Параметр.Форма = ЭтаФорма;
	КонецЕсли;
	
КонецПроцедуры

Процедура ДокументСписокПриПолученииДанных(Элемент, ОформленияСтрок)

		Если КонтекстЭДО.ЗапросыПоКоторымПолученыОтветы <> Неопределено И КонтекстЭДО.ЗапросыПоКоторымПолученыОтветы.Найти(ОформлениеСтроки.ДанныеСтроки.Ссылка) <> Неопределено Тогда
			ОформлениеСтроки.Шрифт = Новый Шрифт(ОформлениеСтроки.Шрифт, , , Истина);
		КонецЕсли;
	КонецЦикла;
